Write a C program to check whether a year is leap year or not:


Answer:

      
(01)

      #include
        int main(){
        int y;
        printf("Enter The Year= ");
        scanf("%d",&y);
        if((y%400==0)||(y%100!=0)&&(y%4==0)){
          printf("%d is a Leap Year",y);
        }
        else{
          printf("%d is not a Leap Year",y);
        }
        return 0;
        }
