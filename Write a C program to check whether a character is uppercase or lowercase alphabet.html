Write a C program to check whether a character is uppercase or lowercase alphabet:


Answer:

    #include
       int main(){
       char ch; 
       printf("Enter The Character= ");
       scanf("%c",&ch);
       if(ch>='a'&&ch<='z'){
                    printf("%c is a lowercase alphabet",ch);
                }
       else if(ch>='A'&&ch<='Z'){
                    printf("%c is an uppercase alphabet",ch);
                }
       else{
                    printf("invalid");
                }
       return 0;
                }
